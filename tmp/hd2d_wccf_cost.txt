	function do_time(time) {
	  var units = [
		{ name: "second", limit: 60, in_seconds: 1 },
		{ name: "minute", limit: 3600, in_seconds: 60 },
		{ name: "hour", limit: 86400, in_seconds: 3600  },
		{ name: "day", limit: 604800, in_seconds: 86400 },
		{ name: "week", limit: 2629743, in_seconds: 604800  },
		{ name: "month", limit: 31556926, in_seconds: 2629743 },
		{ name: "year", limit: null, in_seconds: 31556926 }
	  ];
	  var diff = (new Date() - new Date(time*1000)) / 1000;
	  if (diff < 5) return "now";
	  
	  var i = 0, unit;
	  while (unit = units[i++]) {
		if (diff < unit.limit || !unit.limit){
		  var diff =  Math.floor(diff / unit.in_seconds);
		  return diff + " " + unit.name + (diff>1 ? "s" : "") + " ago";
		}
	  };
	}

	function do_search(index, area, keyword) {

		const searchClient = algoliasearch('JP6Z5G2P43', '4784f5ddc534ec220ede8a86867f33c7');

		const search = instantsearch({
			searchClient,
			indexName: index,
			routing: true,
			initialUiState: {
				wp_searchable_posts: {
					query: keyword
				},
				wp_searchable_posts_recent: {
					query: keyword
				}
			}
		});		

		const renderHits = (renderOptions, isFirstRender) => {

			const { hits, widgetParams } = renderOptions;

			widgetParams.container.innerHTML = `
				
				${hits
				.map(
					item =>
					`<div class="post">
						<div class="post-thumb">
							<img width="564" height="317" src="${(item.images.thumbnail) ? item.images.thumbnail.url : ''}" alt="post-thumb" loading="lazy">
						</div>
						<div class="post-content">
							<h3><a href="${item.permalink}">${item.post_title}</a></h3>
							<footer>
								<span class="post-author">${item.post_author.display_name}</span>
								<span class="post-time">${do_time(item.post_date)}</span>
							</footer>
						</div>
					</div>`
				)
				.join('')}
				
				`;
		};

		const customHits = instantsearch.connectors.connectHits(renderHits);

		

		search.addWidget(
			customHits({
				container: document.querySelector('#' + area),
			})
		);
		
		search.start();
	}
